i. Find the average salary of instructors in those departments where the average salary is more than Rs. 42000.
mysql> SELECT AVG(salary) AS avg_salary 
       FROM instructor 
       WHERE dept_name IN (
           SELECT dept_name 
           FROM instructor 
           GROUP BY dept_name 
           HAVING AVG(salary) > 42000
       );

ii. Increase the salary of each instructor in the Computer department by 10%.
mysql> UPDATE instructor 
       SET salary = salary * 1.10 
       WHERE dept_name = 'Computer';

iii. Find the names of instructors whose names are neither ‘Amol’ nor ‘Amit’.
mysql> SELECT name 
       FROM instructor 
       WHERE name NOT IN ('Amol', 'Amit');

iv. Find the names of students that contain ‘am’ as a substring.
mysql> SELECT name 
       FROM student 
       WHERE name LIKE '%am%';

v. Find the names of students from the Computer department that take “DBMS” courses.
mysql> SELECT s.name 
       FROM student s 
       JOIN course c ON s.dept_name = c.dept_name 
       WHERE c.title = 'DBMS' AND s.dept_name = 'Computer';
